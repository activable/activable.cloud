1. Get the application URL by running these commands:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "activable.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  You can watch the status by running:
    kubectl get svc -w {{ include "activable.fullname" . }} --namespace {{ .Release.Namespace }}

  export SERVICE_IP=$(kubectl get svc {{ include "activable.fullname" . }} --namespace {{ .Release.Namespace }} --template "{.status.loadBalancer.ingress[0].ip}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else }}
  kubectl port-forward svc/{{ include "activable.fullname" . }} 8080:{{ .Values.service.port }} --namespace {{ .Release.Namespace }}
  echo http://localhost:8080
{{- end }}

2. Access the GraphQL Playground:
  {{- if eq .Values.service.type "ClusterIP" }}
  After port-forwarding, open http://localhost:8080 in your browser
  {{- else }}
  Open http://{{ include "activable.fullname" . }}.{{ .Release.Namespace }}:{{ .Values.service.port }} in your browser
  {{- end }}

3. Check the GraphQL endpoint:
  curl http://{{ include "activable.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local:8080/graphql

4. Monitor logs:
  kubectl logs -f deployment/{{ include "activable.fullname" . }} --namespace {{ .Release.Namespace }}

5. For local development with Kind/k3d:
  kind load docker-image localhost:5000/activable:latest --name activable
  helm install activable . -f values-local.yaml
